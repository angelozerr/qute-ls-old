BNF for QUTEParser

NON-TERMINALS
Expression ::= <C_IDENTIFIER> | <STRING_LITERAL> | <NUMBER> | Parenthesis
Parenthesis ::= <OPEN_PAREN> Expression <CLOSE_PAREN>
Interpolation ::= <OPEN_CURLY> Expression <CLOSE_CURLY>
IfBlock ::= <IF> Expression <CLOSE_CURLY> Block ( ElseIfBlock )* ( ElseBlock )? ( <ENDIF> | <ABBREVIATED_END> )
ElseIfBlock ::= <ELSEIF> Expression <CLOSE_CURLY> Block
ElseBlock ::= <ELSE> Block
Section ::= <START_SECTION> ( Expression )? ( <CLOSE_EMPTY> | ( <CLOSE_CURLY> Block ( <END_SECTION> | <ABBREVIATED_END> ) ) )
Block ::= ( <TEXT> | Interpolation | IfBlock | Section )+
Root ::= ( Block )? <EOF>

TOKENS

<QUTE_EXPRESSION> SKIP : { 
<(" " | "\t" | "\n" | "\r")+>
}


<QUTE_EXPRESSION> TOKEN : { 
<C_IDENTIFIER: ("_" | <ASCII_LETTER>) ("_" | <ASCII_LETTER> | <ASCII_DIGIT>)*>
}


<QUTE_EXPRESSION> TOKEN : { 
<STRING_LITERAL: "\"" (<REGULAR_CHAR> | <ESCAPE2> | <ESCAPE1>)* "\"">
|<NUMBER: ("-")? (<ZERO> | <NON_ZERO>) (<FRACTION>)? (<EXPONENT>)?>
}


<QUTE_EXPRESSION> TOKEN : { 
<OPEN_PAREN: "(">
|<CLOSE_PAREN: ")">
}


<QUTE_TEXT> TOKEN : { 
<TEXT: (<NO_OPEN_CURLY> | <ESCAPED_OPEN_CURLY>)+>
|<OPEN_CURLY: "{">
|<IF: "{#if" <WS>>
|<ENDIF: "{/if}">
|<ABBREVIATED_END: "{/}">
|<ELSEIF: "{#else" <WS> "if" <WS>>
|<ELSE: "{#else" (<WS>)? "}">
|<START_SECTION: "{#" (<ASCII_LETTER>)+>
|<END_SECTION: "{/" (<ASCII_LETTER>)+ (<WS>)? "}">
}


<QUTE_TEXT> MORE : { 
<OPEN_COMMENT: "{!">
}


<IN_COMMENT> MORE : { 
<~[]>
}


<IN_COMMENT> SPECIAL_TOKEN : { 
<CLOSE_COMMENT: "!}">
}


<QUTE_EXPRESSION> TOKEN : { 
<CLOSE_CURLY: "}">
|<CLOSE_EMPTY: "/}">
}
